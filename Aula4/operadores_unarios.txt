//OPERADORES UNÁRIOS
void main() {
  
  // Operadores Unários
  int a = 3;
  int b = 4;

  a++;
  --a;
  print(a);
  print(b);
  print(a++ == --b);
  print(a);
  print(b);
  print(--a == --b);
  print(a);
  print(b);

  // Operador Lógico Unário (NOT)

  print(!true);
  print(!false);

  bool x = false;
  print(!x);
}
